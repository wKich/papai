// .opencode/plugins/opencode-tps-meter/src/validation.ts
const DANGEROUS_KEYS = new Set(['__proto__', 'constructor', 'prototype'])

const CONFIG_ALLOWLIST = new Set([
  'enabled',
  'updateIntervalMs',
  'rollingWindowMs',
  'showAverage',
  'showInstant',
  'showTotalTokens',
  'showElapsed',
  'format',
  'minVisibleTPS',
  'fallbackTokenHeuristic',
  'enableColorCoding',
  'slowTpsThreshold',
  'fastTpsThreshold',
])

export function sanitizeConfigKeys(obj: Record<string, unknown>): Record<string, unknown> {
  const result: Record<string, unknown> = {}
  for (const key of Object.keys(obj)) {
    if (!DANGEROUS_KEYS.has(key)) {
      result[key] = obj[key]
    }
  }
  return result
}

export function validateConfigShape(value: unknown): value is Record<string, unknown> {
  if (typeof value !== 'object' || value === null) return false
  if (Array.isArray(value)) return false
  if (Object.getPrototypeOf(value) !== Object.prototype) return false
  for (const key of Object.keys(value as Record<string, unknown>)) {
    if (!CONFIG_ALLOWLIST.has(key)) return false
  }
  return true
}

export function parseFiniteInt(value: string): number | null {
  const trimmed = value.trim()
  if (trimmed === '') return null
  const n = Number(trimmed)
  if (!Number.isFinite(n)) return null
  if (!Number.isInteger(n)) return null
  if (n < 0) return null
  return n
}

export function parseFiniteFloat(value: string): number | null {
  const trimmed = value.trim()
  if (trimmed === '') return null
  const n = Number(trimmed)
  if (!Number.isFinite(n)) return null
  if (n < 0) return null
  return n
}

// .opencode/plugins/opencode-tps-meter/src/sanitize.ts
const ANSI_PATTERN = /\x1b\[[0-9;]*[A-Za-z]/g
const CONTROL_CHAR_PATTERN = /[\x00-\x08\x0b\x0c\x0e-\x1f]/g
const MAX_DISPLAY_LENGTH = 500
const TRUNCATION_INDICATOR = '\u2026'

export function sanitizeForDisplay(input: string): string {
  let cleaned = input.replace(ANSI_PATTERN, '')
  cleaned = cleaned.replace(CONTROL_CHAR_PATTERN, '')
  if (cleaned.length > MAX_DISPLAY_LENGTH) {
    cleaned = cleaned.slice(0, MAX_DISPLAY_LENGTH) + TRUNCATION_INDICATOR
  }
  return cleaned
}

// .opencode/plugins/opencode-tps-meter/src/bounded-map.ts
export type BoundedMapMode = 'monitor' | 'enforce'

export interface BoundedMapOptions<V> {
  mode: BoundedMapMode
  warnThreshold?: number
  maxEntries?: number
  onWarning?: (key: string, size: number) => void
}

export class BoundedMap<K, V> {
  private store = new Map<K, Map<string, V>>()
  private mode: BoundedMapMode
  private warnThreshold: number
  private maxEntries: number
  private onWarning?: (key: string, size: number) => void

  constructor(options: BoundedMapOptions<V>) {
    this.mode = options.mode
    this.warnThreshold = options.warnThreshold ?? 500
    this.maxEntries = options.maxEntries ?? 1000
    this.onWarning = options.onWarning
  }

  get(key: K): Map<string, V> | undefined {
    return this.store.get(key)
  }

  getOrCreate(key: K): Map<string, V> {
    let inner = this.store.get(key)
    if (!inner) {
      inner = new Map<string, V>()
      this.store.set(key, inner)
    }
    return inner
  }

  set(key: K, inner: Map<string, V>): void {
    this.store.set(key, inner)
    this.enforceBounds(key, inner)
  }

  delete(key: K): boolean {
    return this.store.delete(key)
  }

  clear(): void {
    this.store.clear()
  }

  private enforceBounds(key: K, inner: Map<string, V>): void {
    if (this.mode === 'monitor') {
      if (inner.size > this.warnThreshold) {
        this.onWarning?.(String(key), inner.size)
      }
      return
    }

    if (inner.size > this.maxEntries) {
      const excess = inner.size - this.maxEntries
      const keysToDelete: string[] = []
      let count = 0
      for (const k of inner.keys()) {
        if (count >= excess) break
        keysToDelete.push(k)
        count++
      }
      for (const k of keysToDelete) {
        inner.delete(k)
      }
    }
  }
}

// .opencode/plugins/opencode-tps-meter/src/timer-registry.ts
type TimerHandle = ReturnType<typeof setTimeout>

export class TimerRegistry {
  private timers = new Map<string, TimerHandle>()
  private exitHandler: () => void

  constructor() {
    this.exitHandler = () => this.clearAll()
    process.on('exit', this.exitHandler)
  }

  register(label: string, handle: TimerHandle): void {
    const existing = this.timers.get(label)
    if (existing !== undefined) {
      clearTimeout(existing)
    }
    this.timers.set(label, handle)
  }

  clear(label: string): void {
    const handle = this.timers.get(label)
    if (handle !== undefined) {
      clearTimeout(handle)
      this.timers.delete(label)
    }
  }

  has(label: string): boolean {
    return this.timers.has(label)
  }

  clearAll(): void {
    for (const handle of this.timers.values()) {
      clearTimeout(handle)
    }
    this.timers.clear()
  }

  destroy(): void {
    this.clearAll()
    process.removeListener('exit', this.exitHandler)
  }
}

```typescript
// .opencode/plugins/opencode-tps-meter/build.ts
import path from 'path'
import { createHash } from 'crypto'
import { readFileSync, writeFileSync, existsSync, unlinkSync } from 'fs'

async function build(): Promise<void> {
  const srcDir = path.join(import.meta.dir, 'src')
  const outDir = path.join(import.meta.dir, 'dist')

  const esmResult = await Bun.build({
    entrypoints: [path.join(srcDir, 'index.ts')],
    outdir: outDir,
    format: 'esm',
    naming: { entry: '[name].mjs' },
    target: 'node',
    external: ['@opencode-ai/plugin', 'zod'],
    minify: false,
    splitting: false,
  })

  if (!esmResult.success) {
    console.error('ESM build failed:', esmResult.logs)
    process.exit(1)
  }

  const cjsResult = await Bun.build({
    entrypoints: [path.join(srcDir, 'index.ts')],
    outdir: outDir,
    format: 'cjs',
    naming: { entry: '[name].js' },
    target: 'node',
    external: ['@opencode-ai/plugin', 'zod'],
    minify: false,
    splitting: false,
  })

  if (!cjsResult.success) {
    console.error('CJS build failed:', cjsResult.logs)
    process.exit(1)
  }

  const cjsPath = path.join(outDir, 'index.js')
  let cjsContent = readFileSync(cjsPath, 'utf-8')

  const originalLines = cjsContent.split('\n')
  const patchedLines: string[] = []
  let replaced = false

  for (const line of originalLines) {
    if (!replaced && /^module\.exports\s*=\s*__toCommonJS\(/.test(line)) {
      patchedLines.push('')
      replaced = true
    } else {
      patchedLines.push(line)
    }
  }

  if (!replaced) {
    for (const line of originalLines) {
      if (!replaced && /^module\.exports\s*=/.test(line)) {
        patchedLines.push('')
        replaced = true
      }
    }
  }

  patchedLines.push('')
  patchedLines.push('// OpenCode compatibility: export plugin function')
  patchedLines.push('module.exports = exports_src.default;')
  patchedLines.push('module.exports.default = exports_src.default;')
  patchedLines.push('Object.defineProperty(module.exports, "__esModule", { value: true });')
  patchedLines.push('')

  cjsContent = patchedLines.join('\n')

  const exportCount = (cjsContent.match(/^module\.exports\s*=/gm) ?? []).length
  if (exportCount !== 1) {
    console.error(`CJS validation failed: found ${exportCount} module.exports assignments, expected 1`)
    cleanupOutput(outDir)
    process.exit(1)
  }

  const validateProc = Bun.spawn(
    [
      'bun',
      '-e',
      `const m = require("${cjsPath}"); process.exit(typeof m === "function" && typeof m.default === "function" ? 0 : 1)`,
    ],
    { stdout: 'pipe', stderr: 'pipe' },
  )
  const exitCode = await validateProc.exited

  if (exitCode !== 0) {
    console.error('CJS validation failed: exported value is not a function')
    cleanupOutput(outDir)
    process.exit(1)
  }

  writeFileSync(cjsPath, cjsContent)
  console.log('\u2713 Validated CJS exports (function export confirmed)')

  const distPkgPath = path.join(outDir, 'package.json')
  writeFileSync(distPkgPath, JSON.stringify({ type: 'commonjs' }, null, 2))
  console.log('\u2713 Created dist/package.json with type: commonjs')

  const checksums: Record<string, string> = {}
  for (const file of ['index.js', 'index.mjs']) {
    const filePath = path.join(outDir, file)
    if (existsSync(filePath)) {
      const hash = createHash('sha256').update(readFileSync(filePath)).digest('hex')
      checksums[file] = hash
    }
  }
  writeFileSync(path.join(outDir, '.checksums.json'), JSON.stringify(checksums, null, 2) + '\n')
  console.log('\u2713 Recorded build checksums')

  const tscProcess = Bun.spawn(['bunx', 'tsc', '--emitDeclarationOnly', '--declaration', '--outDir', 'dist'], {
    cwd: import.meta.dir,
    stdout: 'inherit',
    stderr: 'inherit',
  })
  const tscExit = await tscProcess.exited
  if (tscExit !== 0) {
    console.error('Type declaration generation failed')
    process.exit(1)
  }

  console.log('\u2713 Build completed successfully')
  console.log('  - dist/index.mjs (ESM)')
  console.log('  - dist/index.js (CommonJS - validated)')
  console.log('  - dist/index.d.ts (TypeScript declarations)')
  console.log('  - dist/.checksums.json (SHA-256 checksums)')
}

function cleanupOutput(outDir: string): void {
  for (const file of ['index.js', 'index.mjs', '.checksums.json']) {
    const p = path.join(outDir, file)
    if (existsSync(p)) {
      try {
        unlinkSync(p)
      } catch {}
    }
  }
}

if (import.meta.main) {
  build().catch((error) => {
    console.error('Build failed:', error)
    process.exit(1)
  })
}

export { build }
```

    exit 1

  if [ "$expected" != "$actual" ]; then
    echo "ERROR: checksum mismatch for $file"
    echo "  expected: $expected"
    echo "  actual:   $actual"
    exit 1
  fi
  echo "  $file: OK"
done

echo "Build verification passed"
